public class StudentCode {
    public static void main(String[] args) {
        // Create an ArrayList for storing class names
        ArrayList<String> classes = new ArrayList<>();
        classes.add("SQL");
        classes.add("C++");
        classes.add("Java");
        classes.add("Precalculus");

        // Make a HashMap to store the student's schedule
        // The key is the day of the week, and the value is an ArrayList of classes on that day.
        HashMap<String, ArrayList<String>> schedule = new HashMap<>();

        // Add Monday classes to the schedule
        ArrayList<String> mondayClasses = new ArrayList<>();
        mondayClasses.add("SQL");
        schedule.put("Monday", mondayClasses);  // Map Monday to its classes

        // Add Tuesday classes to the schedule
        ArrayList<String> tuesdayClasses = new ArrayList<>();
        tuesdayClasses.add("Precalculus");
        schedule.put("Tuesday", tuesdayClasses);  // Map Tuesday to its classes

        // Add Wednesday classes to the schedule
        ArrayList<String> wednesdayClasses = new ArrayList<>();
        wednesdayClasses.add("C++");
        wednesdayClasses.add("SQL");
        schedule.put("Wednesday", wednesdayClasses);  // Map Wednesday to its classes

        // Add Thursday classes to the schedule
        ArrayList<String> thursdayClasses = new ArrayList<>();
        thursdayClasses.add("Precalculus");
        schedule.put("Thursday", thursdayClasses);  // Map Thursday to its classes

        // Add Saturday classes to the schedule
        ArrayList<String> saturdayClasses = new ArrayList<>();
        saturdayClasses.add("Java");
        schedule.put("Saturday", saturdayClasses);  // Map Saturday to its classes

        // Create a HashMap for storing the start times of each class.
        // The key is the class name, and the value is the start time.
        HashMap<String, Integer> startTime = new HashMap<>();
        startTime.put("SQL", 6);          // SQL class starts at 6Am
        startTime.put("C++", 1);          // C++ class starts at 1Pm
        startTime.put("Java", 9);         // Java class starts at 9AM
        startTime.put("Precalculus", 3);  // Precalculus class starts at 3PM

        // Create a Student object with the name "Sadad Yusuf", age 25, and the initialized classes, schedule, and startTime maps.
        Student sadad = new Student("Sadad Yusuf", 25, classes, schedule, startTime);

        // Print out the student's name using the getName() method
        System.out.println("Name: " + sadad.getName());

        // Print out the student's age using the getAge() method
        System.out.println("Age: " + sadad.getAge());

        // Print out the list of classes the student is enrolled in using the getClasses() method
        System.out.println("Classes: " + sadad.getClasses());

        // Print the start time of the student's classes on Monday using the getStartTime() method
        System.out.println("Monday's Class starts at: " + sadad.getStartTime("Monday"));

        // Test for another day (e.g., Wednesday) to see the multiple start times for classes on that day
        System.out.println("Wednesday's Classes start at: " + sadad.getStartTime("Wednesday"));

        // Test for a day with no classes to check the "no classes" output
        System.out.println("Sunday's Class: " + sadad.getStartTime("Sunday"));
    }
}
