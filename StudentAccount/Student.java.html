public class Student {

    // Instance variables representing a student's name, age, classes, schedule, and class start times.
    String name;  // Student's name
    int age;  // Student's age
    ArrayList<String> classes;  // List of all classes the student is enrolled in
    Map<String, ArrayList<String>> schedule;  // Map representing the student's schedule with day of the week as the key and a list of classes for that day
    Map<String, Integer> startTime;  // Map storing the start time for each class with class name as the key

    // Constructor that initializes a Student object with the provided name, age, classes, schedule, and start time information.
    public Student(String name, int age, ArrayList<String> classes, Map<String, ArrayList<String>> schedule, Map<String, Integer> startTime) {
        this.name = name;
        this.age = age;
        this.classes = classes;
        this.schedule = schedule;
        this.startTime = startTime;
    }

    // Getter method that returns the student's name.
    public String getName() {
        return name;
    }

    // Getter method that returns the student's age.
    public int getAge() {
        return age;
    }

    // Getter method that returns the list of classes the student is enrolled in.
    public ArrayList<String> getClasses() {
        return classes;
    }

    /* 
     * Method that takes in a day of the week and returns a list of start times 
     * for all the classes the student has on that day. If the student has no classes on that day, 
     * a message is printed and an empty list is returned. 
     */
    public String getStartTime(String dayOfWeek) {
        // Create an empty list to store the start times for classes on the given day.
        String startTimes = "";

        // Retrieve the list of classes the student has on given day.
        ArrayList<String> classesOnDay = schedule.get(dayOfWeek);

        // If there are classes on the given day, iterate through the list of classes.
        if (classesOnDay != null && !classesOnDay.isEmpty()) {
            for (String className : classesOnDay) {
                // Get the start time of the class from the startTime map.
                Integer theStartTimeIS = startTime.get(className);

                // Add the start time to the list of startTimes.
                startTimes+= theStartTimeIS + ", ";
            }
            if(!startTimes.isEmpty()){
                startTimes = startTimes.substring(0, startTimes.length() - 2);
            }
            return startTimes;
        } else {
            // print a message, for in case no classes on given day.
            return "There are no classes on that day";
        }
    }
}
